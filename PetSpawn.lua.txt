--[[--- Main Logic ---]]
local function showKeyGui()
    local keyGui, keyBox, checkBtn, infoLabel = createKeySystemGui()
    keyGui.Parent = LocalPlayer.PlayerGui

    checkBtn.MouseButton1Click:Connect(function()
        local keyInput = keyBox.Text
        if keyInput == "Nhanhub" then
            keyGui:Destroy()
            showSpawnerGui()
        else
            infoLabel.Text = "Sai key! Key phải là Nhanhub."
        end
    end)
end

local function showSpawnerGui()
    local spawnerGui, nameBox, ageBox, kgBox, spawnBtn, infoLabel = createSpawnerGui()
    spawnerGui.Parent = LocalPlayer.PlayerGui

    spawnBtn.MouseButton1Click:Connect(function()
        local petName = nameBox.Text
        local petAge = ageBox.Text
        local petKg = kgBox.Text

        if petName == "" or petAge == "" or petKg == "" then
            infoLabel.Text = "Vui lòng nhập đủ thông tin pet!"
            infoLabel.TextColor3 = Color3.new(1,0.1,0.1)
            return
        end

        -- Replace this section with your actual pet-spawning logic!
        infoLabel.Text = "Đã spawn pet: "..petName.." | Tuổi: "..petAge.." | Kg: "..petKg
        infoLabel.TextColor3 = Color3.new(0.1,1,0.1)

        -- Example: If you have a RemoteEvent for spawning pets:
        -- ReplicatedStorage:FindFirstChild("SpawnPetEvent"):FireServer(petName, petAge, petKg)
    end)
end
